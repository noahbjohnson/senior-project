    Inequality_WIID,
    Inequality_WDI,

    Economic_MADDISON,
    Education_WDI,
    Labor_WDI,
    Urbanization_UNPD],how="left").join(Economic_WDI.drop(['SI.POV.GINI', 'SI.POV.NAGP'],axis='columns'))
